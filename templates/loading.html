توجه داشته باشید که این قابلیت، توسط پرووایدر theFramework فراخوانی می‌شود. پس، theFramework باید به کنترلری که قصد نمایش toast را دارد پاس داده شود. همچنین دایرکتیو tf-components باید در صفحه موجود باشد. 

//html
<tf-components>
</tf-components>

//js
function Controller($theFramework){
    /*
        $theFramework.loading(showOrHide, [duration]);
    */
    $theFramework.loading(true, 3500); // show loading for 3.5 secs.
    $theFramework.loading(true); // show loading forever.
    $theFramework.loading(hide); // hide every opened loadings.
}
