اتریبیوت‌های ng-model و ng-change هم همانطور که از نامشان پیداست، برای دریافت مدل و فانکشن هنگام تغییر استفاده می‌شود. 

//html
<tf-switch class="form-control" ng-model="switch" ng-change="onChange">
</tf-switch>

//js
switch = ''; /* switch value */
onChange = function(){
    alert('date changed!');
}
